module top;
	reg [7:0] arr_a;
	reg [7:0] arr_b;
	parameter integer c = 0;
	initial begin
		arr_a = 8'hff;
		arr_b = 8'h00;
		$display(":assert: (('%h' == 'ff') and ('%h' == '00'))", arr_a, arr_b);
		arr_b[4+:c] = arr_a[1+:c];
		$display(":assert: ('%b' == '00000000')", arr_b);
	end
endmodule
