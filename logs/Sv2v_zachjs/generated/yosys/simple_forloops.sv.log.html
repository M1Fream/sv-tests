module forloops01 (
	clk,
	a,
	b,
	p,
	q,
	x,
	y
);
	input clk;
	input a;
	input b;
	output reg [3:0] p;
	output reg [3:0] q;
	output reg [3:0] x;
	output reg [3:0] y;
	integer k;
	always @(posedge clk) begin
		for (k = 0; (k < 2); k = (k + 1))
			p[(2 * k)+:2] = ({a, b} ^ {2 {k}});
		x <= (k + {a, b});
	end
	always @(*) begin
		for (k = 0; (k < 4); k = (k + 1))
			q[k] = ({(~ a), (~ b), a, b} >> k[1:0]);
		y = (k - {a, b});
	end
endmodule
